    **Create src/components/pitch/remotion-feature.tsx:**
    Client component ("use client") that wraps Remotion's Player for embedding compositions:
    ```typescript
    "use client";
    import { Player } from "@remotion/player";
    import { useCallback } from "react";

    type CompositionName = "player" | "browse" | "paywall" | "dashboard";

    const compositionMap: Record<CompositionName, { width: number; height: number; duration: number }> = {
      player: { width: 400, height: 700, duration: 150 },
      browse: { width: 1280, height: 720, duration: 180 },
      paywall: { width: 800, height: 600, duration: 180 },
      dashboard: { width: 1280, height: 720, duration: 180 },
    };

    export function RemotionFeature({ composition }: { composition: CompositionName }) {
      const config = compositionMap[composition];
      const lazyComponent = useCallback(
        () => import(`./remotion/${composition}-demo`),
        [composition]
      );
      return (
        <Player
          lazyComponent={lazyComponent}
          durationInFrames={config.duration}
          fps={30}
          compositionWidth={config.width}
          compositionHeight={config.height}
          style={{ width: "100%", borderRadius: 12 }}
          loop
          autoPlay
          controls={false}
        />
      );
    }
    ```
    CRITICAL: This must be a client component with "use client" directive. The Player component uses browser APIs. Use `lazyComponent` with `useCallback` for code splitting (per research pattern 5).

    If Remotion Player causes SSR issues, wrap the entire component with `next/dynamic` and `ssr: false` in the parent that uses it. Add a comment noting this possibility.
